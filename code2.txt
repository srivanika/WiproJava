import java.io.*;
import java.util.*;

// Read only region start
class UserMainCode
{
    public int addLastDigits(int input1, int input2)
    {
        // Read only region end

        input1 = Math.abs(input1);
        input2 = Math.abs(input2);

        return (input1 % 10) + (input2 % 10);
    }
}
